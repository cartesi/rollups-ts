# listOutputs

## Usage

```ts twoslash
import { http } from "viem";
import { createCartesiPublicClient } from "@cartesi/viem";

const publicClientL2 = createCartesiPublicClient({
  transport: http("http://127.0.0.1:6751/rpc"),
});

const { data, pagination } = await publicClientL2.listOutputs({
  application: "0x...",
  limit: 10,
  offset: 0,
  // epochIndex, inputIndex, outputType, voucherAddress (optional)
});
```
