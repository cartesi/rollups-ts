# waitForInput

## Usage

```ts twoslash
import { http } from "viem";
import { createCartesiPublicClient } from "@cartesi/viem";

const publicClientL2 = createCartesiPublicClient({
  transport: http("http://127.0.0.1:6751/rpc"),
});

const input = await publicClientL2.waitForInput({
  application: "0x...",
  inputIndex: 1n,
  // waitProcessing, rejectErrors, pollingInterval, retryCount, timeout (optional)
});
```
